public class Account {
     private int accountID;
    private Enum accountType;
    private double balance;
    private String customerName;

    public Account(int accountID, Enum accountType, double balance, String customerName) {
        this.accountID = accountID;
        this.accountType = accountType;
        this.balance = balance;
        this.customerName = customerName;
    }

    @Override
    public String toString() {
        return "Account{" + "accountID=" + accountID + ", accountType=" + accountType + ", balance=" + balance + ", customerName=" + customerName + '}';
    }
    
    
    public void deposit(double amount) {
        this.balance = amount + this.balance;
        System.out.println("Your remaining balance is " + this.balance);
    }

    public void withdraw(double amount) {
        if(amount < this.balance){
            this.balance = this.balance - amount;
            System.out.println("Your remaining balance is " + this.balance);
        }
        else
            System.out.println("Sorry , not enough money for this transaction");
    }
    
    public void checkBalance(){
        System.out.println("Your  balance is " + this.balance);
    }
    
    public void setBalance(double amount){
        this.balance = amount;
        System.out.println("Your  balance is " + this.balance);
    }

    public int getAccountID() {
        return accountID;
    }

    public void setAccountID(int accountID) {
        this.accountID = accountID;
    }

    public Enum getAccountType() {
        return accountType;
    }

    public void setAccountType(Enum accountType) {
        this.accountType = accountType;
    }

    public double getBalance() {
        return balance;
    }


    public String getCustomerName() {
        return customerName;
    }

    public void setCustomerName(String customerName) {
        this.customerName = customerName;
    }
}
